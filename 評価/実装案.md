````AI2
private record SourceFormat(string Ext, string Vcodec, string Acodec);

private static SourceFormat? ReadSourceFormat(string path)
{
    try
    {
        if (!File.Exists(path)) return null;
        var line = File.ReadAllLines(path).FirstOrDefault(l => !string.IsNullOrWhiteSpace(l));
        if (string.IsNullOrEmpty(line)) return null;

        var parts = line.Split('|');
        return new SourceFormat(
            Ext:    parts.Length > 0 ? parts[0].Trim() : null,
            Vcodec: parts.Length > 1 ? parts[1].Trim() : "none",
            Acodec: parts.Length > 2 ? parts[2].Trim() : "none");
    }
    catch { return null; }
}

### 3. 実際のフォーマット検証 — ffprobe ラッパー

`YtDlpDownloader` に新しいメソッドを追加します。ffprobe は ffmpeg と同ディレクトリに配置されている前提です。

```csharp
private record ActualFormat(string? Ext, string? Vcodec, string? Acodec);

private async Task<ActualFormat?> ProbeFileFormatAsync(string? filePath, string? ffmpegPath)
{
    if (string.IsNullOrEmpty(filePath) || !File.Exists(filePath))
        return null;

    // ffprobe のパスを解決（ffmpeg と同ディレクトリ）
    string? ffprobePath = ResolveFfprobePath(ffmpegPath);
    if (string.IsNullOrEmpty(ffprobePath) || !File.Exists(ffprobePath))
    {
        // ffprobe がない場合は拡張子のみ返す
        var ext = Path.GetExtension(filePath).TrimStart('.').ToLowerInvariant();
        return new ActualFormat(ext, null, null);
    }

    try
    {
        // JSON形式でフォーマット情報を取得
        var psi = new ProcessStartInfo
        {
            FileName = ffprobePath,
            Arguments = $"-v error -select_streams v:0 -show_entries stream=codec_name -of csv=p=0 \"{filePath}\"",
            RedirectStandardOutput = true,
            UseShellExecute = false,
            CreateNoWindow = true
        };
        using var p1 = Process.Start(psi);
        var vcodec = (await p1!.StandardOutput.ReadToEndAsync()).Trim();
        await p1!.WaitForExitAsync(CancellationToken.None);

        // 音声ストリーム
        psi.Arguments = $"-v error -select_streams a:0 -show_entries stream=codec_name -of csv=p=0 \"{filePath}\"";
        using var p2 = Process.Start(psi);
        var acodec = (await p2!.StandardOutput.ReadToEndAsync()).Trim();
        await p2!.WaitForExitAsync(CancellationToken.None);

        var ext = Path.GetExtension(filePath).TrimStart('.').ToLowerInvariant();
        return new ActualFormat(
            ext,
            string.IsNullOrEmpty(vcodec) ? "none" : vcodec,
            string.IsNullOrEmpty(acodec) ? "none" : acodec);
    }
    catch
    {
        var ext = Path.GetExtension(filePath).TrimStart('.').ToLowerInvariant();
        return new ActualFormat(ext, null, null);
    }
}

private static string? ResolveFfprobePath(string? ffmpegPath)
{
    if (string.IsNullOrEmpty(ffmpegPath)) return null;
    var dir = Path.GetDirectoryName(ffmpegPath);
    if (string.IsNullOrEmpty(dir)) return null;
    return Path.Combine(dir, "ffprobe.exe");
}
```

```csharp
/// <summary>
/// yt-dlp / ffprobe のコーデック名を正規化して比較可能にする。
/// </summary>
private static string NormalizeCodec(string? codec)
{
    if (string.IsNullOrWhiteSpace(codec)) return "none";
    var c = codec.ToLowerInvariant().Trim();

    // 動画コーデック
    if (c.StartsWith("avc1") || c == "h264") return "h264";
    if (c.StartsWith("av01") || c == "av1") return "av1";
    if (c.StartsWith("vp9")) return "vp9";
    if (c.StartsWith("vp8")) return "vp8";

    // 音声コーデック
    if (c.StartsWith("mp4a") || c == "aac") return "aac";
    if (c == "opus") return "opus";
    if (c == "vorbis") return "vorbis";
    if (c == "mp3") return "mp3";
    if (c.StartsWith("ac-3") || c == "ac3") return "ac3";

    if (c == "none" || c == "null") return "none";
    return c;
}

#### 4b. 比較とログ出力

```csharp
private void CompareAndLogFormats(DownloadJob job)
{
    var jobLabel = YtDlpFailureFormatter.BuildJobLabel(job);
    var requestedFormat = YtDlpArgumentBuilder.NormalizeFormat(job.Format);
    var isAudioOnly = YtDlpArgumentBuilder.IsAudioFormat(requestedFormat);

    var srcV = NormalizeCodec(job.SourceVcodec);
    var srcA = NormalizeCodec(job.SourceAcodec);
    var actV = NormalizeCodec(job.ActualVcodec);
    var actA = NormalizeCodec(job.ActualAcodec);

    // コンテナの比較（ソース拡張子 ≠ 実ファイﾙ拡張子）
    // ※ mp3/wav 音声抽出は元が m4a/opus 等なので不一致は正常
    bool containerMismatch = !string.IsNullOrEmpty(job.SourceExt)
        && !string.IsNullOrEmpty(job.ActualExt)
        && !job.SourceExt.Equals(job.ActualExt, StringComparison.OrdinalIgnoreCase);

    // 動画コーデックの比較
    bool videoMismatch = !isAudioOnly
        && srcV != "none" && actV != "none"
        && srcV != actV;

    // 音声コーデックの比較
    // ※ mp3/wav は再エンコードが必須のため、音声コーデック不一致は正常とする
    bool audioMismatch = !isAudioOnly
        && srcA != "none" && actA != "none"
        && srcA != actA;

    // mp3/wav/m4a の音声抽出は変換が前提なので警告対象外
    bool isExpectedConversion = isAudioOnly && (requestedFormat == "mp3" || requestedFormat == "wav");
    bool isMismatch = (containerMismatch && !isExpectedConversion) || videoMismatch || audioMismatch;

    job.HasFormatMismatch = isMismatch;

    // ログ出力
    var sourceDesc = $"{job.SourceExt ?? "?"}";
    var actualDesc = $"{job.ActualExt ?? "?"}";

    if (!string.IsNullOrEmpty(job.SourceVcodec) && job.SourceVcodec != "none")
        sourceDesc += $"/{NormalizeCodec(job.SourceVcodec)}";
    if (!string.IsNullOrEmpty(job.SourceAcodec) && job.SourceAcodec != "none")
        sourceDesc += $"/{NormalizeCodec(job.SourceAcodec)}";

    if (!string.IsNullOrEmpty(job.ActualVcodec) && job.ActualVcodec != "none")
        actualDesc += $"/{NormalizeCodec(job.ActualVcodec)}";
    if (!string.IsNullOrEmpty(job.ActualAcodec) && job.ActualAcodec != "none")
        actualDesc += $"/{NormalizeCodec(job.ActualAcodec)}";

    if (string.IsNullOrEmpty(job.SourceExt) && string.IsNullOrEmpty(job.ActualExt))
    {
        _logger.Info($"フォーマット確認 {jobLabel} / 情報取得不可（スキップ）");
        return;
    }

    if (isMismatch)
    {
        var reasons = new List<string>();
        if (containerMismatch && !isExpectedConversion)
            reasons.Add($"コンテナ不一致({job.SourceExt}→{job.ActualExt})");
        if (videoMismatch)
            reasons.Add($"動画codec不一致({srcV}→{actV})");
        if (audioMismatch)
            reasons.Add($"音声codec不一致({srcA}→{actA})");

        var reason = string.Join(", ", reasons);
        _logger.Warn($"フォーマット不一致 {jobLabel} / ソース={sourceDesc} → ダウンロード={actualDesc} / {reason}");

        job.FormatMismatchTooltip = $"フォーマットが一致しません\n" +
            $"ソース: {sourceDesc}\n" +
            $"ダウンロード: {actualDesc}\n" +
            $"差異: {reason}";
    }
    else
    {
        _logger.Info($"フォーマット一致 {jobLabel} / ソース={sourceDesc} → ダウンロード={actualDesc}");

        // 音声変換があった場合は情報としてツールチップに出す（警告ではない）
        if (isExpectedConversion && containerMismatch)
        {
            job.FormatMismatchTooltip = $"音声変換: {job.SourceExt}/{srcA} → {job.ActualExt}";
        }
    }
}

### 7. 考慮事項

| 項目 | 対応 |
| --- | --- |
| **音声のみ(mp3/wav)** | 再エンコードが必須のため、コンテナ・音声codecの不一致は**警告対象外**とする。ログには INFO で出力し、ツールチップに「音声変換」として情報表示 |
| **m4a** | 通常コピーのため不一致は出ない。もし不一致が出たら異常として警告 |
| **ffprobe 未インストール** | 拡張子のみで比較（コーデック比較はスキップ）。ログに「ffprobe未検出、拡張子のみ比較」と注記 |
| **`--print-to-file` が空** | yt-dlp のバージョンや動画によって出力されない場合がある。その場合は「情報取得不可」としてスキップ |
| **フォールバック再試行時** | 初回失敗 → 再試行で成功した場合、再試行時の `video:` フェーズ出力で上書きされるため、最終的なソースフォーマットが正しく記録される |
| **マージ（動画+音声）** | `--merge-output-format mp4` で webm→mp4 のリマックスが発生した場合、コーデックは同じだがコンテナが変わる。「コンテナ不一致」として警告（実害は低いが透明性のため） |
| **パフォーマンス** | ffprobe は数100msで終了するため、ダウンロード完了後の待ち時間には影響しない |
````
